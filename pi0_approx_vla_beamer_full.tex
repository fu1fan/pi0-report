      \small

\section{Architecture Bottleneck}

\begin{frame}
  \frametitle{任务一：FPGA 部署瓶颈总览}
  \small
  \begin{columns}[T,onlytextwidth]
    \column{0.49\textwidth}
    \begin{block}{不是单一算子问题}
      \piZero 的瓶颈来自完整推理链路叠加：
      \begin{itemize}
        \item GB 级模型权重与片外访存；
        \item 多视角视觉 token 膨胀；
        \item Transformer GEMM / Attention 的高计算量；
        \item 中间 activation / KV cache / attention map 占用。
      \end{itemize}
    \end{block}

    \column{0.47\textwidth}
    \begin{block}{VLA 额外困难}
      与普通 VLM 相比，机器人动作链路还引入：
      \begin{itemize}
        \item state/action projection 与 action expert；
        \item flow matching 多步迭代；
        \item 连续动作对低比特定点误差更敏感；
        \item 50Hz 控制场景对端到端延迟要求更高。
      \end{itemize}
    \end{block}
  \end{columns}

  \vspace{4pt}
  \begin{alertblock}{核心结论}
    完整 \piZero 不适合直接部署到资源受限 FPGA；应抽取 projector、Linear/GEMM、attention softmax、norm/activation、action expert 等模块做近似计算与 HLS kernel 验证。
  \end{alertblock}
\end{frame}

\begin{frame}
  \frametitle{瓶颈数据：存储与序列长度量级}
  \small
  \begin{columns}[T,onlytextwidth]
    \column{0.45\textwidth}
    \begin{block}{3.3B 参数权重存储估算}
      \centering
      \scriptsize
      \setlength{\tabcolsep}{5pt}
      \begin{tabular}{@{}lrr@{}}
        \toprule
        精度        & Bytes/param & 权重规模    \\
        \midrule
        FP32      & 4.0         & 13.2 GB \\
        FP16/BF16 & 2.0         & 6.6 GB  \\
        INT8      & 1.0         & 3.3 GB  \\
        INT4      & 0.5         & 1.65 GB \\
        \bottomrule
      \end{tabular}
    \end{block}

    \begin{alertblock}{存储含义}
      \footnotesize
      即使 INT8 量化，完整权重仍是 GB 级；FPGA 片上存储更适合缓存 tile、activation 和 LUT，而不是完整模型权重。
    \end{alertblock}

    \column{0.51\textwidth}
    \begin{block}{视觉 token 与 attention score 估算}
      \centering
      \scriptsize
      \setlength{\tabcolsep}{3pt}
      \begin{tabular}{@{}lrrr@{}}
        \toprule
        场景      & tokens & $T^2$ score & 相对 224 单图     \\
        \midrule
        224 单图  & 256    & 65K         & 1.0$\times$   \\
        224 三相机 & 768    & 589K        & 9.0$\times$   \\
        448 单图  & 1024   & 1.05M       & 16.0$\times$  \\
        448 三相机 & 3072   & 9.44M       & 144.0$\times$ \\
        896 单图  & 4096   & 16.78M      & 256.0$\times$ \\
        \bottomrule
      \end{tabular}
    \end{block}

    \begin{exampleblock}{序列长度含义}
      \footnotesize
      token 数上升不仅增加视觉前端计算，还会按 $T^2$ 放大 attention score、softmax 和片上缓存压力。
    \end{exampleblock}
  \end{columns}
\end{frame}

\begin{frame}
  \frametitle{瓶颈定位：哪些模块最值得近似计算}
  \footnotesize
  \begin{table}
    \setlength{\tabcolsep}{3pt}
    \begin{tabular}{@{}p{0.24\linewidth}p{0.42\linewidth}p{0.25\linewidth}@{}}
      \toprule
      模块                     & FPGA 部署瓶颈                  & 近似优化切入点               \\
      \midrule
      Q/K/V/O Linear           & 大量矩阵乘，DSP 与权重带宽占用高         & INT8 / W4A8 tiled GEMM \\
      Attention Score $QK^T$ & 序列长度增加后呈二次复杂度              & 分块、低比特 score        \\
      Softmax                & exp、sum、division 对硬件不友好    & LUT、PWL、Taylor、base-2 \\
      FFN / MLP              & Transformer 中最大 GEMM 来源之一 & INT8 GEMM、低秩/剪枝       \\
      GELU / SiLU            & tanh / sigmoid 等非线性成本高       & LUT / PWL / hard approximation \\
      RMSNorm                & 涉及平方、均值、倒数平方根            & fixed-point + rsqrt          \\
      Flow Matching           & action expert 多步重复调用        & 减少 Euler steps \\
      \bottomrule
    \end{tabular}
  \end{table}

  \vspace{3pt}
  \begin{alertblock}{本项目策略}
    先用 PyTorch 判断误差风险，再把有价值的算子落到 Vitis HLS，得到 latency、II、LUT、FF、BRAM、DSP 等硬件指标。
  \end{alertblock}
\end{frame}

\section{Approximation Strategy}

\begin{frame}
  \frametitle{任务二：面向 \piZero 的模块级近似计算方案}
  \small
  \begin{table}
    \scriptsize
    \setlength{\tabcolsep}{3pt}
    \begin{tabular}{@{}p{0.22\linewidth}p{0.40\linewidth}p{0.28\linewidth}@{}}
      \toprule
      优化对象               & 近似/定点化方法                                             & 预期收益                  \\
      \midrule
      Linear / QKV / FFN & INT8、INT4、W4A8 fake quant；INT32 累加后 requantize & 降低 GEMM 计算和权重存储       \\
      Softmax            & LUT、PWL、Taylor、base-2 softmax                        & 避免高成本 exp/div         \\
      GELU / SiLU        & PWL、hard approximation、clipped linear                & 降低复杂激活函数开销            \\
      RMSNorm            & fixed-point + reciprocal sqrt 近似                     & 降低 sqrt/div 成本        \\
      Flow steps         & 10 steps $\rightarrow$ 8/6/4/2 steps                 & 减少 action expert 调用次数 \\
      \bottomrule
    \end{tabular}
  \end{table}

  \vspace{4pt}
  \begin{exampleblock}{FPGA 适配}
    Vitis HLS 的 arbitrary precision 和 fixed-point 类型可以直接指定位宽，便于实现 INT8/INT16、LUT exp、PWL GELU 和 rsqrt 近似。
  \end{exampleblock}
\end{frame}

\section{PyTorch Results}

\begin{frame}
  \frametitle{PyTorch 结果一：GEMM / Projector / FFN 优先采用 INT8}
  \small
  \begin{columns}[T,onlytextwidth]
    \column{0.58\textwidth}
    \begin{table}
      \scriptsize
      \setlength{\tabcolsep}{3pt}
      \begin{tabular}{@{}p{0.34\linewidth}p{0.26\linewidth}cc@{}}
        \toprule
        实验阶段 & 覆盖对象 & INT8 最低 cosine & INT4 最低 cosine \\
        \midrule
        random shape & Linear / projector & $\approx$0.999 & $\approx$0.988--0.990 \\
        \piZero-aligned random & QKV / FFN / action expert & 0.998402 & 0.963150 \\
        real \piZero weight & 真实 projector / FFN / projection & 0.994843 & 0.921080 \\
        \bottomrule
      \end{tabular}
    \end{table}

    \begin{exampleblock}{观察}
      \footnotesize
      INT8 在三类实验中均保持较高 cosine similarity；INT4 / W4A8 虽然显著降低存储，但真实权重下误差明显扩大。
    \end{exampleblock}

    \column{0.38\textwidth}
    \begin{alertblock}{工程结论}
      \footnotesize
      主方案：INT8 tiled GEMM。\vspace{2pt}

      适用模块：visual projector、QKV/O projection、FFN、state/action projection、action expert MLP。\vspace{2pt}

      INT4/W4A8：作为激进压缩方向保留，不作为主结果。
    \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}
  \frametitle{PyTorch 结果二：Softmax / RMSNorm 可近似，激活替换需谨慎}
  \small
  \begin{columns}[T,onlytextwidth]
    \column{0.32\textwidth}
    \begin{block}{Softmax}
      \footnotesize
      输入先 subtract max，并对近似路径做 clamp。真实 Q/K score 实验中，LUT softmax 是最好的 hardware-like 方案之一，适合转 HLS LUT kernel。
    \end{block}

    \column{0.32\textwidth}
    \begin{block}{RMSNorm}
      \footnotesize
      真实权重实验中 RMSNorm 近似最大 relative L2 约为 0.0117，说明 fixed-point + rsqrt 方向误差可控。
    \end{block}

    \column{0.32\textwidth}
    \begin{alertblock}{GELU / SiLU}
      \footnotesize
      identity 或过激 clipped-linear 替换可能让 relative L2 接近 1。激活函数不能只看单点形式，必须放回完整 FFN 输出评价。
    \end{alertblock}
  \end{columns}

  \vspace{4pt}
  \begin{exampleblock}{结论}
    Softmax 与 RMSNorm 的近似比较可控；GELU/SiLU 要优先采用保守 PWL/LUT，不建议直接替换成 identity 或过粗 clipped-linear。
  \end{exampleblock}
\end{frame}

\begin{frame}
  \frametitle{PyTorch 结果三：Flow Step Reduction 是算法级加速旋钮}
  \small
  \begin{columns}[T,onlytextwidth]
    \column{0.50\textwidth}
    \begin{block}{实验设置}
      \begin{itemize}
        \item \piZero 链路：noisy action \ra multi-step flow integration \ra clean action chunk；
        \item toy flow matching：$H=50$, action dim $=32$, cond dim $=1024$；
        \item 对比 10 / 8 / 6 / 4 / 2 step Euler integration。
      \end{itemize}
    \end{block}

    \column{0.46\textwidth}
    \begin{exampleblock}{观察}
      \footnotesize
      2-step 相比 10-step 约 $4.46\times$ 加速；相对 10-step 输出的 MSE 约 $4.32\times10^{-3}$，cosine 约 0.9975。
    \end{exampleblock}

    \begin{alertblock}{限制}
      \footnotesize
      这部分说明减少 flow step 有潜在加速收益。真实动作质量仍需要 observation / action trajectory 和任务成功率验证。
    \end{alertblock}
  \end{columns}
\end{frame}

  \end{exampleblock}

\section{Conclusion}

\begin{frame}
  \frametitle{结论与边界}
  \small
  \begin{enumerate}
    \item \kw{架构层面：} \piZero 是 VLM backbone + action expert + flow matching，完整上板不现实，应拆成可优化模块。
    \item \kw{PyTorch 层面：} 已完成 random shape、\piZero-aligned random、real-weight proxy 和 toy flow step reduction；INT8 是当前最稳的 GEMM 类量化方案。
    \item \kw{HLS 层面：} 已实现并综合 INT8 GEMM、LUT softmax、PWL GELU、RMSNorm rsqrt 和 fixed projector tile，获得 latency、II、LUT、FF、BRAM、DSP 数据。
    \item \kw{方案判断：} Softmax/GELU 近似在 HLS 中资源收益明显；RMSNorm 近似误差较小但当前 DSP 代价偏高；INT4/W4A8 和激进 activation replacement 需谨慎。
  \end{enumerate}

  \begin{block}{一句话收束}
    本次工作不是完整复现 \piZero，而是把 VLA 推理链路拆成可在 FPGA 上验证的关键模块，并完成了从 PyTorch 误差评估到 Vitis HLS 可综合报告的闭环验证。
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{后续工作}
  \small
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \begin{block}{短期可补强}

        \item 不能声称真实机器人成功率不下降；
        \item PyTorch fake quant latency 不能等同真实 INT8 硬件加速；
        \item toy flow 只说明 step-latency tradeoff。
      \end{itemize}
    \end{alertblock}

\begin{frame}
  \frametitle{Appendix：答辩问答准备}
  \small
  \begin{block}{Q1：为什么不完整部署 \piZero？}
    \footnotesize
    因为完整模型参数和推理链路远超一周内资源受限 FPGA 部署范围；本项目采用模块级方法，聚焦 projector、GEMM、softmax、activation、norm 和 flow steps，这些模块能对应真实 VLA 推理瓶颈。
  \end{block}

  \begin{block}{Q2：为什么说 INT8 是主方案？}
    \footnotesize
    随机、\piZero-aligned 和真实权重 proxy 都显示 INT8 cosine 较高；INT4/W4A8 压缩更强但真实权重下误差明显上升，因此不作为主结论。
  \end{block}

  \begin{block}{Q3：HLS 结果说明什么？}
    \footnotesize
    说明近似算子不是停留在 PyTorch 层面，而是能综合为 FPGA kernel，并输出 latency、II、LUT、FF、BRAM、DSP 等硬件指标。
  \end{block}
\end{frame}
